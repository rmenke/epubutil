29class ambiguous_option : public option_error {
30 template <class Range>
31 std::string as_enumeration(Range &&r) {
32 if (auto sz = std::ranges::distance(r); sz == 1) {
33 return *std::ranges::begin(r);
34 }
35 else if (sz == 2) {
36 auto iter = std::ranges::begin(r);
37 return *iter + std::string{" and "} + *std::ranges::next(iter);

41 for (auto &&opt : std::ranges::take_view(r, sz - 1)) {
42 result += opt + std::string{", "};
43 }

45 }

169 using namespace std;
170
171 while (first != last) {
172 string_view str = *first;
173
174 if (!str.starts_with('-')) return first;
175 if (str == "--"sv) return next(first);
176

178 std::string arg;
179
180 str.remove_prefix(2);
181

183 arg = str.substr(offset + 1);

185 }

188 auto opt_end = opt_begin;

192 ++opt_end;
193 }

201 std::views::keys |

218 }
219 else {
