33class ambiguous_option : public option_error {
34 template <class Range>
35 std::string as_enumeration(Range &&r) {
36 if (auto sz = std::ranges::distance(r); sz == 1) {
37 return *std::ranges::begin(r);
38 }
39 else if (sz == 2) {
40 auto iter = std::ranges::begin(r);
41 return *iter + std::string{" and "} + *std::ranges::next(iter);
42 }
43 else {
44 std::string result;
45 for (auto &&opt : std::ranges::take_view(r, sz - 1)) {
46 result += opt + std::string{", "};
47 }
48 return result + "and " + *std::ranges::rbegin(r);
49 }
50 }

242 template <class ForwardIt>
243 ForwardIt process(ForwardIt first, ForwardIt last) try {
244 using namespace std;
245
246 while (first != last) {
247 string_view str = *first;
248
249 if (!str.starts_with('-')) return first;
250 if (str == "--"sv) return next(first);
251
252 if (str.starts_with("--"sv)) {
253 std::string arg;
254
255 str.remove_prefix(2);
256
257 if (auto offset = str.find('='); offset != str.npos) {
258 arg = str.substr(offset + 1);
259 str = str.substr(0, offset);
260 }
261
262 auto opt_begin = _long.lower_bound(std::string{str});
263 auto opt_end = opt_begin;
264
265 while (opt_end != _long.end() &&
266 opt_end->first.starts_with(str)) {
267 ++opt_end;
268 }
269
270 if (opt_begin == opt_end) {
271 throw option_not_found("--"s + std::string{str});
272 }
273 if (distance(opt_begin, opt_end) > 1) {
274 throw ambiguous_option(
275 str, std::ranges::subrange(opt_begin, opt_end) |
276 std::views::keys |
277 std::views::transform([](auto &&s) {
278 return std::string{"--"} + s;
279 }));
280 }
281
282 const auto &option = opt_begin->second;
283
284 if (option->needs_arg()) {
285 if (arg.empty()) {
286 if (++first == last) {
287 throw argument_needed("--"s + opt_begin->first);
288 }
289 arg = *first;
290 }
291
292 option->set_arg(arg);
293 }
294 else {
295 if (!arg.empty()) {
296 throw argument_not_needed("--"s + opt_begin->first);
297 }
298
299 option->set();
300 }
301 }
302 else {
303 str.remove_prefix(1);
304 while (!str.empty()) {
305 auto flag = str.front();
306
307 auto found = _short.find(flag);
308 if (found == _short.end()) {
309 throw option_not_found('-' + std::string{flag});
310 }
311
312 const auto &option = found->second;
313
314 if (option->needs_arg()) {
315 auto arg = std::string{str.substr(1)};
316 if (arg.empty()) {
317 if (++first == last) {
318 throw argument_needed(
319 '-' + std::string{found->first});
320 }
321 arg = *first;
322 }
323 option->set_arg(arg);
324 break;
325 }
326 else {
327 option->set();
328 }
329
330 str.remove_prefix(1);
331 }
332 }
333
334 ++first;
335 }
336
337 return first;
338 }
339 catch (const option_error &ex) {
340 if (_passthru) throw;
341 std::cerr << "error: " << ex.what() << "\n\n";
342 usage();
343 exit(1);
344 }

346 template <class StringLike>
347 static auto
348 wrap(StringLike &&s, typename std::decay<StringLike>::type::size_type w,
349 typename std::decay<StringLike>::type::size_type i = 0) {
350 if (s.size() <= w) return std::forward<StringLike>(s);
351
352 auto pos = s.rfind(' ', w);
353 if (pos == s.npos) return s;
354
355 auto end = pos + 1;
356 while (s[pos - 1] == ' ') --pos;
357
358 auto result = std::string(s, 0, pos) + "\n" + std::string(i, ' ');
359 return result + wrap(std::forward<StringLike>(s).substr(end), w, i);
360 }

362 void usage(std::ostream &out = std::cerr,
363 std::size_t screen_width = 72) const {
364 if (!_synopsis.empty()) {
365 out << wrap("usage: " + _synopsis, screen_width, 7) << "\n\n";
366 }
367
368 auto width =
369 std::ranges::max(_usage | std::views::elements<0> |
370 std::views::transform(&std::string::size));
371
372 static constexpr std::string::size_type indent = 4;
373 static constexpr std::string::size_type padding = 8;
374
375 width += indent + padding;
376
377 for (const auto &[flags, text] : _usage) {
378 std::string description(indent, ' ');
379
380 description.append(flags);
381 description.resize(width, ' ');
382 description.append(text);
383

385 }
