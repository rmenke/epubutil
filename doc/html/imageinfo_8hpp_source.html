312 template<typename T>
313 inline T readInt(off_t offset, bool swapEndian = false) {
314 T val = *((T *) (data() + offset));
315 return swapEndian ? swapE<T>(val) : val;
316 }
317
318 inline std::string readString(off_t offset, size_t size) {
319 return std::string((char *) data() + offset, size);
320 }
321
322 inline std::string toString() {
323 return std::string((char *) data(), size());
324 }
325
326 inline bool cmp(off_t offset, size_t size, const void *buf) {
327 return memcmp(data() + offset, buf, size) == 0;
328 }
329
330 inline bool cmpAnyOf(off_t offset, size_t size, const std::initializer_list<const void *> &bufList) {
331 return std::any_of(bufList.begin(), bufList.end(), [this, offset, size](const void *buf) {
332 return memcmp(data() + offset, buf, size) == 0;
333 });
334 }
335
336private:
337 template<typename T>
338 static T swapE(T u) {
339 union {
340 T u;
341 uint8_t u8[sizeof(T)];
342 } source{}, dest{};
343 source.u = u;
344 for (size_t k = 0; k < sizeof(T); k++)
345 dest.u8[k] = source.u8[sizeof(T) - k - 1];
346 return dest.u;
347 }

351 size_t m_size = 0;

356class IIReadInterface {
357public:
358 IIReadInterface() = delete;
359
360 IIReadInterface(IIReadFunc &readFunc, size_t length)
361 : m_readFunc(readFunc),
362 m_length(length) {
363#ifndef II_DISABLE_HEADER_CACHE
364 m_headerCache.alloc(std::min((size_t) II_HEADER_CACHE_SIZE, length));
365 read(m_headerCache.data(), 0, m_headerCache.size());
366#endif
367 }
368
369 inline IIBuffer readBuffer(off_t offset, size_t size) {
370 assert(offset >= 0);
371 assert(offset + size <= m_length);
372 IIBuffer buffer(size);
373#ifndef II_DISABLE_HEADER_CACHE
374 if (offset + size <= m_headerCache.size()) {
375 memcpy(buffer.data(), m_headerCache.data() + offset, size);
376 } else if (offset < m_headerCache.size() && m_headerCache.size() - offset >= (II_HEADER_CACHE_SIZE / 4)) {
377 size_t head = m_headerCache.size() - offset;
378 memcpy(buffer.data(), m_headerCache.data() + offset, head);
379 read(buffer.data() + head, offset + head, size - head);
380 } else {
381 read(buffer.data(), offset, size);
382 }
383#else
384 read(buffer.data(), offset, size);
385#endif
386 return buffer;
387 }
388
389 inline size_t length() const {
390 return m_length;
391 }
392
393private:
394 inline void read(void *buf, off_t offset, size_t size) {
395 m_readFunc(buf, offset, size);
396 }
397
398private:
399 IIReadFunc &m_readFunc;
400 size_t m_length = 0;
401#ifndef II_DISABLE_HEADER_CACHE
402 IIBuffer m_headerCache;
403#endif
404};
